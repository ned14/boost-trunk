Description

Performs short-circuit logical or (||) operation on its arguments.

Example

Find the position of a type in a list? [1]. 

template<
      typename Sequence
    , typename T
    >
struct find_predicate
{
    template< typename Iterator > struct apply
    {
     private:
        typedef boost::is_same<
              Iterator
            , typename end?<Sequence>::type
            > end_of_sequence;


        typedef boost::is_same<typename Iterator::type,T> type_found;


     public:
        typedef logical not< logical or<
             end_of_sequence
           , type_found
           > > type;
    };
};




typedef list?<int,char,float,double,long,double,float> list;
typedef while_true?<
      find_predicate<list,double>
    , begin?<list>::type
    , next<_1>
    >::type iter;


BOOST_STATIC_ASSERT((
    distance?< iter, begin?<list>::type >::value == 3
    ));

Notes

[1] This is an example of how to use the short-circuit logic operations to simplify your code; it is not the recommended way of searching for something in a list? or any other Sequence. The find or find if algorithms offer a better way of doing it.
