# Helper macro to create tests for the iostreams library
macro(iostreams_test TESTNAME)
  boost_test_run(${TESTNAME} 
    ${ARGN}
    DEPENDS boost_unit_test_framework
    COMPILE_FLAGS "-DBOOST_IOSTREAMS_NO_LIB")
endmacro(iostreams_test)
